// Configuraci√≥n de CLABE - REEMPLAZA con tu CLABE interbancaria real
const CLABE_INTERBANCARIA = '638180000176182306';
const form = document.getElementById('setupForm');
const shareSection = document.getElementById('shareSection');
const shareLink = document.getElementById('shareLink');
const copyBtn = document.getElementById('copyBtn');
const coffeeBtn = document.getElementById('coffeeBtn');
const modal = document.getElementById('clabeModal');
const closeBtn = document.querySelector('.modal-close');
const clabeDisplay = document.getElementById('clabeDisplay');
const copyClabeBtn = document.getElementById('copyClabeBtn');
// Configurar modal CLABE
if (clabeDisplay) {
clabeDisplay.textContent = CLABE_INTERBANCARIA;
}
// Abrir modal al hacer click en el bot√≥n
if (coffeeBtn) {
coffeeBtn.addEventListener('click', () => {
modal.classList.add('active');
});
}
// Cerrar modal al hacer click en la X
if (closeBtn) {
closeBtn.addEventListener('click', () => {
modal.classList.remove('active');
});
}
// Cerrar modal al hacer click fuera del contenido
if (modal) {
modal.addEventListener('click', (e) => {
if (e.target === modal) {
modal.classList.remove('active');
}
});
}
// Cerrar modal con tecla ESC
document.addEventListener('keydown', (e) => {
if (e.key === 'Escape' && modal && modal.classList.contains('active')) {
modal.classList.remove('active');
}
});
// Copiar CLABE al portapapeles
if (copyClabeBtn) {
copyClabeBtn.addEventListener('click', () => {
if (navigator.clipboard && navigator.clipboard.writeText) {
navigator.clipboard.writeText(CLABE_INTERBANCARIA).then(() => {
copyClabeBtn.textContent = '‚úì Copiado!';
copyClabeBtn.classList.add('copied');
setTimeout(() => {
copyClabeBtn.textContent = 'üìã Copiar CLABE';
copyClabeBtn.classList.remove('copied');
}, 2000);
}).catch(() => {
fallbackCopyTextToClipboard(CLABE_INTERBANCARIA, copyClabeBtn);
});
} else {
fallbackCopyTextToClipboard(CLABE_INTERBANCARIA, copyClabeBtn);
}
});
}
// Funci√≥n fallback para copiar texto
function fallbackCopyTextToClipboard(text, button) {
const textArea = document.createElement('textarea');
textArea.value = text;
textArea.style.position = 'fixed';
textArea.style.left = '-999999px';
textArea.style.top = '-999999px';
document.body.appendChild(textArea);
textArea.focus();
textArea.select();
try {
document.execCommand('copy');
button.textContent = '‚úì Copiado!';
button.classList.add('copied');
setTimeout(() => {
button.textContent = 'üìã Copiar CLABE';
button.classList.remove('copied');
}, 2000);
} catch (err) {
console.error('Error al copiar:', err);
button.textContent = '‚ùå Error';
setTimeout(() => {
button.textContent = 'üìã Copiar CLABE';
}, 2000);
}
document.body.removeChild(textArea);
}

form.addEventListener('submit', (e) => {
e.preventDefault();
const fromName = document.getElementById('fromName').value.trim();
const toName = document.getElementById('toName').value.trim();

if (!fromName || !toName) {
alert('Por favor completa ambos campos');
return;
}

// Reemplazar TODOS los espacios con guiones (m√∫ltiples espacios se convierten en un solo gui√≥n)
let fromNameClean = fromName.trim().replace(/\s+/g, '-');
let toNameClean = toName.trim().replace(/\s+/g, '-');
// Asegurar que no queden espacios de ning√∫n tipo
fromNameClean = fromNameClean.replace(/\s/g, '-').replace(/[%]/g, '');
toNameClean = toNameClean.replace(/\s/g, '-').replace(/[%]/g, '');
// Crear URL en formato dominio/valentine/
const baseUrl = window.location.origin;
// Usar /valentine/ como ruta base
const pathBase = '/valentine/v2/';
// Construir la URL en formato dominio/valentine/msg.html?from=...&to=...
const shareUrl = baseUrl + pathBase + 'msg.html?from=' + fromNameClean + '&to=' + toNameClean;
// Mostrar la URL en el input
shareLink.value = shareUrl;
shareSection.style.display = 'block';
// NO redirigir, solo mostrar la URL para copiar
});

copyBtn.addEventListener('click', () => {

document.execCommand('copy');

copyBtn.textContent = originalText;

}, 2000);
});
